
CREATE SCHEMA DB_SENSO;
USE DB_SENSO;
CREATE TABLE CATEGORIA
(
	CD_CATEGORIA INT,
	NM_CATEGORIA VARCHAR(24),
	CONSTRAINT PK_CATEGORIA PRIMARY KEY (CD_CATEGORIA)
);
CREATE TABLE NOTICIA
(
	CD_NOTICIA INT,
	NM_TITULO_NOTICIA VARCHAR(80),
	NM_LINHA_FINA_NOTICIA VARCHAR(150),
	DS_TEXTO_NOTICIA TEXT,
	DT_PUBLICACAO_NOTICIA DATE,
	HR_PUBLICACAO_NOTICIA TIME,
	IC_DESTAQUE_NOTICIA BOOL,
    CD_CATEGORIA INT(11),
	CONSTRAINT PK_NOTICIA PRIMARY KEY(CD_NOTICIA),
	CONSTRAINT FK_CATEGORIA FOREIGN KEY (CD_CATEGORIA) REFERENCES CATEGORIA(CD_CATEGORIA)
);
CREATE TABLE AUTOR
(
	CD_AUTOR INT,
	NM_AUTOR VARCHAR(45),
	CONSTRAINT PK_AUTOR PRIMARY KEY (CD_AUTOR)
);
CREATE TABLE AUTOR_NOTICIA
(
	CD_AUTOR INT,
	CD_NOTICIA INT,
	CONSTRAINT PK_AUTOR_NOTICIA PRIMARY KEY (CD_NOTICIA, CD_AUTOR),
	CONSTRAINT FK_NA_AUTOR FOREIGN KEY (CD_AUTOR) REFERENCES AUTOR(CD_AUTOR),
	CONSTRAINT FK_NA_NOTICIA FOREIGN KEY(CD_NOTICIA) REFERENCES NOTICIA(CD_NOTICIA)
);
CREATE TABLE BANNER
(
	CD_BANNER INT,
	NM_EMPRESA_ANUNCIANTE_BANNER VARCHAR(45),
	IC_ATIVO_BANNER BOOL,
	DT_PUBLICACAO_BANNER DATE,
	HR_PUBLICACAO_BANNER TIME,
	CD_ORDENACAO_BANNER INT,
	NM_LINK_BANNER VARCHAR(255),
	CONSTRAINT PK_BANNER PRIMARY KEY (CD_BANNER)
);
CREATE TABLE TIPO_USUARIO
(
	CD_TIPO_USUARIO INT,
	NM_TIPO_USUARIO VARCHAR(45),
	CONSTRAINT PK_TIPO_USUARIO PRIMARY KEY(CD_TIPO_USUARIO)
);
CREATE TABLE USUARIO
(
	CD_LOGIN_USUARIO VARCHAR(45),
	NM_USUARIO VARCHAR(45),
	NM_SENHA_USUARIO VARCHAR(64),
	CD_TIPO_USUARIO INT,
	CONSTRAINT PK_USUARIO PRIMARY KEY(CD_LOGIN_USUARIO),
	CONSTRAINT FK_USUARIO_TU FOREIGN KEY(CD_TIPO_USUARIO) REFERENCES TIPO_USUARIO(CD_TIPO_USUARIO)
	
);
